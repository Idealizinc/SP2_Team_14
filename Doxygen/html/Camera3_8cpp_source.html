197 void Camera3::rotateCamera(double dt)
198 {
199  //Lock the cursor's position to the center of the screen.
200  int moveX = S_Width / 2;
201  int moveY = S_Height / 2;
202  POINT mousePosition;
203  GetCursorPos(&mousePosition);
204  SetCursorPos(moveX, moveY);
205  CameraYrotation -= ((mousePosition.x - moveX) * (float)(dt)* rotate_Sensitivity);
206  CameraXrotation += ((mousePosition.y - moveY) * (float)(dt)* rotate_Sensitivity);
207  CameraXrotation = Math::Clamp(CameraXrotation, minCameraXrotation, maxCameraXrotation);
208  if (maxCameraYrotation < 360 && minCameraYrotation < 360)
209  {
210  CameraYrotation = Math::Clamp(CameraYrotation, minCameraYrotation, maxCameraYrotation);
211  }
212  target = Vector3(sin(Math::DegreeToRadian(CameraYrotation)) * cos(Math::DegreeToRadian(CameraXrotation)) + position.x,
213  -sin(Math::DegreeToRadian(CameraXrotation)) + position.y,
214  cos(Math::DegreeToRadian(CameraYrotation)) * cos(Math::DegreeToRadian(CameraXrotation)) + position.z);
215  view = (target - position).Normalized();
216  Vector3 right = view.Cross(defaultUp);
217  up = right.Cross(view);
218 }
