493 {
494  if (!mesh || mesh->textureID <= 0) //Proper error check

496 
497  glDisable(GL_DEPTH_TEST);

499  glUniform3fv(m_parameters[U_TEXT_COLOR], 1, &color.r);
500  glUniform1i(m_parameters[U_LIGHTENABLED], 0);
501  glUniform1i(m_parameters[U_COLOR_TEXTURE_ENABLED], 1);
502  glActiveTexture(GL_TEXTURE0);
503  glBindTexture(GL_TEXTURE_2D, mesh->textureID);
504  glUniform1i(m_parameters[U_COLOR_TEXTURE], 0);
505  for (unsigned i = 0; i < text.length(); ++i)
506  {
507  Mtx44 characterSpacing;
508  characterSpacing.SetToTranslation(i * 0.5f, 0, 0); //1.0f is the spacing of each character, you may change this value
509  Mtx44 MVP = projectionStack.Top() * viewStack.Top() * modelStack.Top() * characterSpacing;
510  glUniformMatrix4fv(m_parameters[U_MVP], 1, GL_FALSE, &MVP.a[0]);
511 
512  mesh->Render((unsigned)text[i] * 6, 6);
513  }
514  glBindTexture(GL_TEXTURE_2D, 0);
515  glUniform1i(m_parameters[U_TEXT_ENABLED], 0);
516  glEnable(GL_DEPTH_TEST);
