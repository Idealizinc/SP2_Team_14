110 ik_s32 CIrrKlangAudioStreamMP3::readFrames(void* target, ik_s32 frameCountToRead)
111 {
112  const int frameSize = Format.getFrameSize();
113 
114  int framesRead = 0;
115  ik_u8* out = (ik_u8*)target;
116 
117  while (framesRead < frameCountToRead)
118  {
119  // no more samples? ask the MP3 for more
120  if (DecodedQueue.getSize() < frameSize)
121  {
122  if (!decodeFrame() || EndOfFileReached)
123  return framesRead;
124 
125  // if the buffer is still empty, we are done
126  if (DecodedQueue.getSize() < frameSize)
127  return framesRead;
128  }
129 
130  const int framesLeft = frameCountToRead - framesRead;
131  const int dequeSize = DecodedQueue.getSize() / frameSize;
132  const int framesToRead = framesLeft < dequeSize ? framesLeft : dequeSize;
133 
134  DecodedQueue.read(out, framesToRead * frameSize);
135 
136  out += framesToRead * frameSize;
137  framesRead += framesToRead;
138  Position += framesToRead;
139  }
140 
141  return framesRead;
142 }

146 bool CIrrKlangAudioStreamMP3::decodeFrame()
147 {
148  int outputSize = 0;
149 
150  while (!outputSize)
151  {
152  if (InputPosition == InputLength)
153  {
154  InputPosition = 0;
155  InputLength = File->read(InputBuffer, IKP_MP3_INPUT_BUFFER_SIZE);
156 
157  if (InputLength == 0)
158  {
159  EndOfFileReached = true;
160  return true;
161  }
162  }
163 
164  int rv = mpaudec_decode_frame( TheMPAuDecContext, (ik_s16*)DecodeBuffer,
165  &outputSize,
166  (ik_u8*)InputBuffer + InputPosition,
167  InputLength - InputPosition);
168 
169  if (rv < 0)
170  return false;
171 
172  InputPosition += rv;
173  } // end while
174 
175  if (!FirstFrameRead)
176  {
177  Format.ChannelCount = TheMPAuDecContext->channels;
178  Format.SampleRate = TheMPAuDecContext->sample_rate;
179  Format.SampleFormat = ESF_S16;
180  Format.FrameCount = -1; // unknown lenght
181 
182  FirstFrameRead = true;
183  }
184  else
185  if (TheMPAuDecContext->channels != Format.ChannelCount ||
186  TheMPAuDecContext->sample_rate != Format.SampleRate)
187  {
188  // Can't handle format changes mid-stream.
189  return false;
190  }
191 
192  if (!TheMPAuDecContext->parse_only)
193  {
194  if (outputSize < 0)
195  {
196  // Couldn't decode this frame. Too bad, already lost it.
197  // This should only happen when seeking.
198 
199  outputSize = TheMPAuDecContext->frame_size;
200  memset(DecodeBuffer, 0, outputSize * Format.getFrameSize());
201  }
202 
203  DecodedQueue.write(DecodeBuffer, outputSize);
204  }
205 
206  return true;
207 }

215 bool CIrrKlangAudioStreamMP3::setPosition(ik_s32 pos)
216 {
217  if (!File || !TheMPAuDecContext)
218  return false;
219 
220  if (pos == 0)
221  {
222  // usually done for looping, just reset to start
223 
224  File->seek(FileBegin); // skip possible ID3 header
225 
226  EndOfFileReached = false;
227 
228  DecodedQueue.clear();
229 
230  MPAuDecContext oldContext = *TheMPAuDecContext;
231 
232  mpaudec_clear(TheMPAuDecContext);
233  mpaudec_init(TheMPAuDecContext);
234 
235  TheMPAuDecContext->bit_rate = oldContext.bit_rate;
236  TheMPAuDecContext->channels = oldContext.channels;
237  TheMPAuDecContext->frame_size = oldContext.frame_size;
238  TheMPAuDecContext->sample_rate = oldContext.sample_rate;
239 
240  InputPosition = 0;
241  InputLength = 0;
242  Position = 0;
243  CurrentFramePosition = 0;
244 
245  return true;
246  }
247  else
248  {
249  // user wants to seek in the stream, so do this here
250 
251  int scan_position = 0;
252  int target_frame = 0;
253  int frame_count = (int)FramePositionData.size();
254 
255  while (target_frame < frame_count)
256  {
257  int frame_size = FramePositionData[target_frame].size;
258 
259  if (pos <= scan_position + frame_size)
260  break;
261  else
262  {
263  scan_position += frame_size;
264  target_frame++;
265  }
266  }
267 
268 
269  const int MAX_FRAME_DEPENDENCY = 10;
270  target_frame = std::max(0, target_frame - MAX_FRAME_DEPENDENCY);
271  setPosition(0);
272 
273  File->seek(FramePositionData[target_frame].offset, false);
274 
275  int i;
276  for (i = 0; i < target_frame; i++)
277  {
278  if (i>=(int)FramePositionData.size())
279  {
280  // internal error
281  setPosition(0);
282  return false;
283  }
284 
285  Position += FramePositionData[i].size;
286  }
287 
288  if (!decodeFrame() || EndOfFileReached)
289  {
290  setPosition(0);
291  return false;
292  }
293 
294  int frames_to_consume = pos - Position; // PCM frames now
295  if (frames_to_consume > 0)
296  {
297  ik_u8 *buf = new ik_u8[frames_to_consume * Format.getFrameSize()];
298  readFrames(buf, frames_to_consume);
299  delete[] buf;
300  }
301 
302  return true;
303  }
304 
305  return false;
306 }

366 void CIrrKlangAudioStreamMP3::skipID3IfNecessary()
367 {
368  char header[10];
369  int read = File->read(&header, 10);
370 
371  if (read == 10 &&
372  header[0] == 'I' && header[1] == 'D' && header[2] == '3')
373  {
374  int versionMajor = header[3];
375  int versionMinor = header[4];
376  int flags = header[5];
377 
378  // IDv2 size looks like the following: ID3v2 size 4 * %0xxxxxxx.
379  // Sick, but that's how it works.
380 
381  int size = 0;
382  size = (header[6] & 0x7f) << (3*7);
383  size |= (header[7] & 0x7f) << (2*7);
384  size |= (header[8] & 0x7f) << (1*7);
385  size |= (header[9] & 0x7f) ;
386 
387  size += 10; // header size
388 
389  FileBegin = size;
390  File->seek(FileBegin);
391  }
392  else
393  File->seek(0);
394 }
