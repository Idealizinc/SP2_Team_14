38 unsigned int show_bits(const GetBitContext *s, int n)
39 {
40  int i;
41  unsigned int result = 0;
42  assert(s->size_in_bits - s->index >= n);
43  for (i = s->index; i < s->index + n; i++) {
44  int byte_index = i / 8;
45  unsigned int right_shift = 7 - (i % 8);
46  uint8_t byte = s->buffer[byte_index];
47  uint8_t bit;
48  result <<= 1;
49  if (right_shift == 0)
50  bit = byte & 0x1;
51  else
52  bit = (byte >> right_shift) & 0x1;
53  result |= (unsigned int)bit;
54  }
55  return result;
56 }

96 static int alloc_table(VLC *vlc, int size)
97 {
98  int index;
99  index = vlc->table_size;
100  vlc->table_size += size;
101  if (vlc->table_size > vlc->table_allocated) {
102  vlc->table_allocated += (1 << vlc->bits);
103  vlc->table = realloc(vlc->table,
104  sizeof(VLC_TYPE) * 2 * vlc->table_allocated);
105  if (!vlc->table)
106  return -1;
107  }

111 static int build_table(VLC *vlc, int table_nb_bits,
112  int nb_codes,
113  const void *bits, int bits_wrap, int bits_size,
114  const void *codes, int codes_wrap, int codes_size,
115  uint32_t code_prefix, int n_prefix)
116 {
117  int i, j, k, n, table_size, table_index, nb, n1, index;
118  uint32_t code;
119  VLC_TYPE (*table)[2];
120 
121  table_size = 1 << table_nb_bits;
122  table_index = alloc_table(vlc, table_size);
123 #ifdef DEBUG_VLC
124  printf("new table index=%d size=%d code_prefix=%x n=%d\n",
125  table_index, table_size, code_prefix, n_prefix);
126 #endif
127  if (table_index < 0)
128  return -1;
129  table = &vlc->table[table_index];
130 
131  for(i=0;i<table_size;i++) {
132  table[i][1] = 0; /*bits*/
133  table[i][0] = -1; /*codes*/
134  }
135 
136  /* first pass: map codes and compute auxillary table sizes */
137  for(i=0;i<nb_codes;i++) {
138  GET_DATA(n, bits, i, bits_wrap, bits_size);
139  GET_DATA(code, codes, i, codes_wrap, codes_size);
140  /* we accept tables with holes */
141  if (n <= 0)
142  continue;
143 #if defined(DEBUG_VLC) && 0
144  printf("i=%d n=%d code=0x%x\n", i, n, code);
145 #endif
146  /* if code matches the prefix, it is in the table */
147  n -= n_prefix;
148  if (n > 0 && (code >> n) == code_prefix) {
149  if (n <= table_nb_bits) {
150  /* no need to add another table */
151  j = (code << (table_nb_bits - n)) & (table_size - 1);
152  nb = 1 << (table_nb_bits - n);
153  for(k=0;k<nb;k++) {
154 #ifdef DEBUG_VLC
155  printf("%4x: code=%d n=%d\n",
156  j, i, n);
157 #endif
158  assert(table[j][1] /*bits*/ == 0);
159  table[j][1] = n; /*bits*/
160  table[j][0] = i; /*code*/
161  j++;
162  }
163  } else {
164  n -= table_nb_bits;
165  j = (code >> n) & ((1 << table_nb_bits) - 1);
166 #ifdef DEBUG_VLC
167  printf("%4x: n=%d (subtable)\n",
168  j, n);
169 #endif
170  /* compute table size */
171  n1 = -table[j][1]; /*bits*/
172  if (n > n1)
173  n1 = n;
174  table[j][1] = -n1; /*bits*/
175  }
176  }
177  }
178 
179  /* second pass : fill auxillary tables recursively */
180  for(i=0;i<table_size;i++) {
181  n = table[i][1]; /*bits*/
182  if (n < 0) {
183  n = -n;
184  if (n > table_nb_bits) {
185  n = table_nb_bits;
186  table[i][1] = -n; /*bits*/
187  }
188  index = build_table(vlc, n, nb_codes,
189  bits, bits_wrap, bits_size,
190  codes, codes_wrap, codes_size,
191  (code_prefix << table_nb_bits) | i,
192  n_prefix + table_nb_bits);
193  if (index < 0)
194  return -1;
195  /* note: realloc has been done, so reload tables */
196  table = &vlc->table[table_index];
197  table[i][0] = index; /*code*/
198  }
199  }
200  return table_index;
201 }

204 /* Build VLC decoding tables suitable for use with get_vlc().
205 
206  'nb_bits' set thee decoding table size (2^nb_bits) entries. The
207  bigger it is, the faster is the decoding. But it should not be too
208  big to save memory and L1 cache. '9' is a good compromise.
209 
210  'nb_codes' : number of vlcs codes
211 
212  'bits' : table which gives the size (in bits) of each vlc code.
213 
214  'codes' : table which gives the bit pattern of of each vlc code.
215 
216  'xxx_wrap' : give the number of bytes between each entry of the
217  'bits' or 'codes' tables.
218 
219  'xxx_size' : gives the number of bytes of each entry of the 'bits'
220  or 'codes' tables.
221 
222  'wrap' and 'size' allows to use any memory configuration and types
223  (byte/word/long) to store the 'bits' and 'codes' tables.
224 */
225 int init_vlc(VLC *vlc, int nb_bits, int nb_codes,
226  const void *bits, int bits_wrap, int bits_size,
227  const void *codes, int codes_wrap, int codes_size)
228 {
229  vlc->bits = nb_bits;
230  vlc->table = NULL;
231  vlc->table_allocated = 0;
232  vlc->table_size = 0;
233 #ifdef DEBUG_VLC
234  printf("build table nb_codes=%d\n", nb_codes);
235 #endif
236 
237  if (build_table(vlc, nb_bits, nb_codes,
238  bits, bits_wrap, bits_size,
239  codes, codes_wrap, codes_size,
240  0, 0) < 0) {
241  free(vlc->table);
242  return -1;
243  }
244  return 0;
245 }

253 int get_vlc(GetBitContext *s, const VLC *vlc)
254 {
255  int code = 0;
256  int depth = 0, max_depth = 3;
257  int n, index, bits = vlc->bits;
258 
259  do {
260  index = show_bits(s, bits) + code;
261  code = vlc->table[index][0];
262  n = vlc->table[index][1];
263  depth++;
264 
265  if (n < 0 && depth < max_depth) {
266  skip_bits(s, bits);
267  bits = -n;
268  }
269  } while (n < 0 && depth < max_depth);
270 
271  skip_bits(s, n);
272  return code;
273 }
